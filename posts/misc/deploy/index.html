title: auto git deploy⌗

A simple bash script to push your code to git⌗

# !/bin/bash
set -o nounset
set -o errexit
echo $0
echo $*
echo $$
echo "Will deploy your code to git server..."
git add -A
git commit -m "$1"
git push -u origin $2
echo "Done!"
